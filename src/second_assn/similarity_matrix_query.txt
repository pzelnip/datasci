select A.row_num as row, b.col_num as col, sum(A.value * B.value) as result from A,B where A.col_num = B.row_num group by A.row_num, b.col_num



select A.docid as row, B.term as col, sum(A.count * B.count) as result 
from frequency as A, frequency as B 
where A.docid = B.docid group by A.docid, b.term
